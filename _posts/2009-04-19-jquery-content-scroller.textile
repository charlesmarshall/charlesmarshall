jQuery(document).ready(function() {
	jQuery('#container1').contentscroller({pages:'.cexample'});
	jQuery('#container2').contentscroller({pages:'.cexample', controller:'#controls2', before:"alert('before')", after:"alert('after')"});
});
</script>
